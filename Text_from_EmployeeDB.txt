Departments
-
dept_no varchar pk fk -< Dept_Emp.dept_no fk -< Managers.dept_no
dept_name varchar

Dept_Emp
-
emp_no int pk fk -< Managers.emp_no fk -< Employees.emp_no fk -< Salaries.emp_no fk -< Titles.emp_no
dept_no varchar
from_date date
to_date date

Managers
-
dept_no varchar pk
emp_no int
from_date date
to_date date

Employees
- 
emp_no int pk
birth_date date
first_name varchar
last_name varchar
gender varchar
hire_date date

Salaries
-
emp_no int pk
salary varchar
from_date date
to_date date

Titles
-
emp_no int pk
title varchar
from_date date
to_date date
